Kiểu dữ liệu, tên biến:
01/ biến: biến được hiểu là một vùng nhớ và được gáng một cái value( giá trị) nào đó trogn đoạn code
02/ kiểu dữ liệu: được gáng với một biến thích hợp có các kiểu dữ liệu như:
    + int, short, long: kiểu giá trị nguyên
    + float, double: kiểu số thực có dấu phẩy
    + boolean: true os false
    + char: kiểu ký tự đơn lẻ ''
    + string: kiểu chuổi kỹ tự ""
    + Array: kiểu mảng

03/ cách khai báo một biến:
  type(kiểu dữ liệu mặc định) + varialname(tên biến do mình đặt) = value(giá trị biến);
  vd: int X = 100;
      float Y = 1.5;
      String kH = ABC;


04/ một chương trình C# cơ bản:
 
  using System;        			                // từ khóa using được sử dụng để sử dụng các chức năng của System
  namespace helloW 	{		               // tập hợp các lớp, cụ thể đây là tạo ra lớp helloW chứa class program
         class program{		      // chứa các định ngĩa và phương thức mà chương trình sử dụng ( cụ thể là phương thức Main).
	static void Main(String[] args){   
                                                                              // Main ở đây là một entry point, vùng chứa chương trình thực thi của C#.
	     
	   int X;
	   X =10;
	   string tenHo="Tran Huu Phuc";
	   float diemSo = 10.5;
 	
 	  Console.OutputEncoding = System.Text.Encoding.UTF8;                 // hàm dùng để gõ tiếng việt có dấu

	   Console.WriteLine("gia tri X ={0}", X);
	   Console.WriteLine("Ten ho cua ban la ={0}", tenHo);
	   Console.WriteLine("diem so cua ban la ={0}", diemSo);

	   Console.ReadKey();	             // làm cho chương trình đợi 1 phím được nhấn và không đóng đột ngột
	}
       }
}
